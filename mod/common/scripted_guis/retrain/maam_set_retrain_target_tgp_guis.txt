######################################################
# MAA Manager - TGP MAA Target Setter GUIs
#
# Scripted GUIs for setting retrain targets to TGP MAA types
# Sets maa_manager_retrain_target_type flag and cost variable
######################################################

# ===== CULTURAL MAA =====

maam_set_target_emishi_horse_archers = {
	scope = character
	is_shown = { always = yes }
	effect = {
		set_variable = {
			name = maa_manager_retrain_target_type
			value = flag:emishi_horse_archers
		}
		set_variable = {
			name = maa_manager_retrain_target_cost
			value = light_cavalry_recruitment_cost
		}
	}
}

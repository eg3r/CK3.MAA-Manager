######################################################
# Men-At-Arms Manager - Retrain Target Setter GUIs
#
# Individual scripted GUIs for each MAA type to set the retrain target flag
# Each GUI sets the maa_manager_retrain_target_type scope value on the character
# Scope: Character (player)
######################################################
# Base Game Regular MAA
maam_set_target_light_footmen = {
	scope = character
	effect = {
		save_scope_value_as = {
			name = maa_manager_retrain_target_type
			value = flag:light_footmen
		}
		set_variable = {
			name = maa_manager_retrain_target_type
			value = flag:light_footmen
		}
		set_variable = {
			name = maa_manager_target_cost
			value = skirmisher_recruitment_cost
		}
	}
}
